public class Q8_SentenceAnalysis {
 public static void main(String[] args) {
 String sentence = "Java Programming is FUN and Powerful!";
 // 1. Count characters (including spaces and punctuation)
 int length = sentence.length();
 System.out.println("Character count: " + length);
 // 2. Convert to lowercase
 String lower = sentence.toLowerCase();
 System.out.println("Lowercase: " + lower);
 // 3. Replace spaces with underscores
 String underscored = sentence.replace(' ', '_');
 System.out.println("Spaces replaced: " + underscored);
 // 4. Check whether it contains the word "Java" (case-sensitive example)
 boolean hasJava = sentence.contains("Java");
 System.out.println("Contains 'Java'?: " + hasJava);
 // 5. Display the word "FUN" in lowercase using substring extraction
 int start = sentence.indexOf("FUN");
 if (start != -1) {
 String funLower = sentence.substring(start, start + 3).toLowerCase();
 System.out.println("FUN in lowercase via substring: " + funLower);
 } else {
 System.out.println("'FUN' not found via substring.");
 }
 }
}
